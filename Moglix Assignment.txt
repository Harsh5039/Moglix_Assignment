#include <bits/stdc++.h>
using namespace std;

int main() {
    string s;
    cin >> s;

    stack<int> st;
    st.push(-1);

    int maxLen = 0;
    int endIndex = -1;

    for (int i = 0; i < s.size(); i++) {
        if (s[i] == '(') {
            st.push(i);
        } else {
            st.pop();

            if (st.empty()) {
                st.push(i);
            } else {
                int len = i - st.top();

                if (len > maxLen) {
                    maxLen = len;
                    endIndex = i;
                }
            }
        }
    }

    if (maxLen > 0) {
        cout << s.substr(endIndex - maxLen + 1, maxLen);
    } else {
        cout << "No valid parentheses";
    }

    return 0;
}
